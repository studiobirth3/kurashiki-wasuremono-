📊 Googleフォーム → スプレッドシート自動記録

Googleフォームの送信トリガーに設定してください。管理番号の自動採番・全項目の転記を行います。

// ============================================
// 忘れ物管理 - フォーム送信トリガー
// Googleフォームの「送信時」トリガーに設定
// ============================================

const SHEET_ID = 'YOUR_SPREADSHEET_ID'; // ★ 変更してください
const SHEET_NAME = '忘れ物台帳';

function onFormSubmit(e) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(SHEET_NAME);
  const responses = e.response.getItemResponses();

  // 管理番号の自動採番
  const lastRow = sheet.getLastRow();
  const newNum = String(lastRow).padStart(4, '0');
  const manageId = 'LF-' + Utilities.formatDate(
    new Date(), 'Asia/Tokyo', 'yyyyMMdd'
  ) + '-' + newNum;

  // フォーム回答を取得
  const data = {};
  responses.forEach(r => {
    data[r.getItem().getTitle()] = r.getResponse();
  });

  // スプレッドシートに書き込み
  sheet.appendRow([
    manageId,                          // 管理番号
    new Date(),                        // 受付日時
    data['利用日'] || '',
    data['スタジオ名・部屋名'] || '',
    data['予約名'] || '',
    data['忘れ物の内容'] || '',
    data['色・ブランド・特徴'] || '',
    data['置いた場所'] || '',
    data['連絡先（電話）'] || '',
    data['連絡先（メール）'] || '',
    data['緊急性'] || '通常',
    '',   // 発見日（後で記入）
    '',   // 発見場所（後で記入）
    '',   // 返却方法（後で記入）
    Utilities.formatDate(
      new Date(new Date().getTime() + 14 * 24 * 60 * 60 * 1000),
      'Asia/Tokyo', 'yyyy/MM/dd'
    ),    // 保管期限（14日後）
    '確認中' // ステータス
  ]);

  // 緊急の場合はLINE通知
  if (data['緊急性'] === '緊急') {
    sendLineNotify(manageId, data);
  }
}

const LINE_TOKEN = 'YOUR_LINE_CHANNEL_ACCESS_TOKEN'; // ★
const LINE_USER_ID = 'YOUR_LINE_USER_OR_GROUP_ID';   // ★

function sendLineNotify(manageId, data) {
  const message = [
    '🚨【緊急】忘れ物問い合わせ',
    '─────────────',
    '管理番号: ' + manageId,
    '予約名: ' + (data['予約名'] || '-'),
    'スタジオ: ' + (data['スタジオ名・部屋名'] || '-'),
    '忘れ物: ' + (data['忘れ物の内容'] || '-'),
    '連絡先: ' + (data['連絡先（電話）'] || '-'),
    '─────────────',
    '台帳で確認してください'
  ].join('\n');

  const payload = {
    to: LINE_USER_ID,
    messages: [{ type: 'text', text: message }]
  };

  UrlFetchApp.fetch('https://api.line.me/v2/bot/message/push', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': 'Bearer ' + LINE_TOKEN
    },
    payload: JSON.stringify(payload)
  });
}

      to: LINE_USER_ID,
      messages: [{ type: 'text', text: text }]
    })
  });
}

⏰ 期限チェック（毎日自動実行）

「時間主導のトリガー → 毎日」に設定すると、期限切れ・期限3日前の品をLINE通知します。

// ============================================
// 毎日実行 - 保管期限チェック
// トリガー: 時間主導 → 毎日 → 午前9時
// ============================================

function checkDeadlines() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  const today = new Date();
  today.setHours(0,0,0,0);

  const warnings = [];
  const expired = [];

  // 2行目以降（ヘッダー除く）
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const status = row[15]; // ステータス列
    if (status === '返却済み' || status === '処分済み') continue;

    const deadline = new Date(row[14]); // 保管期限列
    deadline.setHours(0,0,0,0);
    const diffDays = Math.ceil((deadline - today) / (1000*60*60*24));

    if (diffDays < 0) {
      expired.push(row[0] + ' / ' + row[5]); // 管理番号 + 物品名
    } else if (diffDays <= 3) {
      warnings.push(row[0] + ' あと' + diffDays + '日 / ' + row[5]);
    }
  }

  let msg = '';
  if (expired.length > 0) {
    msg += '🔴【期限切れ】処分対象\n' + expired.join('\n') + '\n\n';
  }
  if (warnings.length > 0) {
    msg += '🟡【期限3日以内】要確認\n' + warnings.join('\n');
  }

  if (msg) sendLineMessage('⏰ 保管期限チェック\n─────────────\n' + msg);
}

📷 写真をGoogleドライブに保存するGASコード

ℹ️
スタッフが写真を撮影後、このエンドポイントにBase64データを送るとGoogleドライブに保存されます。スプレッドシートの写真URL列に自動記録されます。
// ============================================
// 写真をGoogleドライブに保存 + 台帳URLを更新
// GASをウェブアプリとして「全員」にデプロイ
// ============================================

const SHEET_ID        = 'YOUR_SPREADSHEET_ID';  // ★ 変更：スプレッドシートのID
const SHEET_NAME      = '忘れ物台帳';               // ★ シート名（変更不要）
const DRIVE_FOLDER_ID = 'YOUR_DRIVE_FOLDER_ID';  // ★ 変更：ドライブフォルダのID

function doPost(e) {
  try {
    // Content-Type: text/plain でも application/json でも対応
    const rawBody = e.postData ? e.postData.contents : '{}';
    const params = JSON.parse(rawBody);
    const base64 = params.imageBase64;   // "data:image/jpeg;base64,..." 形式
    const manageId = params.manageId;    // 管理番号 例: LF-20260401-001
    const mimeType = params.mimeType || 'image/jpeg';

    // Base64 → Blob に変換
    const base64Data = base64.split(',')[1];
    const blob = Utilities.newBlob(
      Utilities.base64Decode(base64Data),
      mimeType,
      manageId + '.jpg'
    );

    // Googleドライブに保存
    const folder = DriveApp.getFolderById(DRIVE_FOLDER_ID);
    const file = folder.createFile(blob);
    file.setSharing(
      DriveApp.Access.ANYONE_WITH_LINK,
      DriveApp.Permission.VIEW
    );
    const fileUrl = 'https://drive.google.com/uc?id=' + file.getId();

    // 台帳スプレッドシートの写真URL列を更新
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sheet = ss.getSheetByName(SHEET_NAME);
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] === manageId) {
        // H列（写真URL列 = 8列目）に書き込み
        sheet.getRange(i + 1, 8).setValue(fileUrl);
        break;
      }
    }

    return ContentService
      .createTextOutput(JSON.stringify({ success: true, url: fileUrl }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

Webアプリ側の送信コード

以下のコードを lost-item-system.html の <script> に追加すると、写真をGASに自動送信してドライブに永続保存できます。

// ★ GASウェブアプリのURLに変更してください
const GAS_ENDPOINT = 'https://script.google.com/macros/s/AKfycbwLgJXkdgBQw3jdgHNIxY3hJYxzwosXjwSaw6ogib9krn6JgAZaj7DIs9LU2gq2c-S5JA/exec';

async function uploadPhotoToGAS(manageId, base64Data) {
  // GASエンドポイント未設定の場合
  if (!hasGasEndpoint()) {
    showToast('⚠️ GASエンドポイント未設定。写真はブラウザ内のみ保存');
    return null;
  }

  showToast('📤 写真をGoogleドライブに送信中...');
  try {
    const res = await fetch(GAS_ENDPOINT, {
      method: 'POST',
      mode: 'cors',
      headers: { 'Content-Type': 'text/plain' }, // GASのCORSに合わせtext/plain
      body: JSON.stringify({
        imageBase64: base64Data,
        manageId: manageId,
        mimeType: 'image/jpeg'
      })
    });
    const text = await res.text();
    let json;
    try { json = JSON.parse(text); } catch(pe) {
      console.error('GASレスポンスのパース失敗:', text);
      showToast('❌ GASからの応答が不正です。デプロイ設定を確認してください');
      return null;
    }
    if (json && json.success) {
      return json.url;
    } else {
      showToast('❌ 保存失敗: ' + (json && json.error ? json.error : '不明なエラー'));
      return null;
    }
  } catch(e) {
    console.error('写真アップロード失敗:', e);
    // CORSエラーの可能性が高い → GASのデプロイ設定を案内
    if (e.name === 'TypeError' && e.message.includes('fetch')) {
      showToast('❌ 通信エラー。GASのデプロイ設定「アクセス：全員」を確認してください');
    } else {
      showToast('❌ 写真アップロード失敗: ' + e.message);
    }
    return null;
  }
}

🔄 台帳データをスプレッドシートと同期するGASコード

ℹ️
このGASをウェブアプリとしてデプロイすると、Webアプリの登録・更新データがスプレッドシートに保存され、どの端末からでも同じデータが見られるようになります。
写真保存のGASと同じプロジェクトに追記してください。
// ============================================
// 台帳データ同期 API
// 写真保存GASと同じプロジェクトに追記する
// ============================================

// doPost に以下のアクションを追加（既存のdoPostを置き換え）
function doPost(e) {
  try {
    const rawBody = e.postData ? e.postData.contents : '{}';
    const params = JSON.parse(rawBody);
    const action = params.action;

    // ── 台帳：1件保存 / 更新 ──
    if (action === 'saveItem') {
      return saveItem(params.item);
    }
    // ── 台帳：全件取得 ──
    if (action === 'getAll') {
      return getAllItems();
    }
    // ── 写真アップロード（既存） ──
    if (action === 'uploadPhoto' || params.imageBase64) {
      return uploadPhoto(params);
    }

    return jsonResponse({ success: false, error: 'unknown action' });
  } catch(err) {
    return jsonResponse({ success: false, error: err.toString() });
  }
}

// ── 全件取得 ──
function getAllItems() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(SHEET_NAME);
  const rows = sheet.getDataRange().getValues();
  if (rows.length <= 1) return jsonResponse({ success: true, items: [] });

  const headers = rows[0];
  const items = rows.slice(1).map(row => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = row[i] || '');
    return obj;
  });
  return jsonResponse({ success: true, items: items });
}

// ── 1件保存 / 更新 ──
function saveItem(item) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(SHEET_NAME);
  const rows = sheet.getDataRange().getValues();

  // ヘッダー行がなければ作成
  if (rows.length === 0) {
    sheet.appendRow([
      'id','inquiryDate','foundDate','place','item','detail',
      'contact','returnMethod','urgency','status','deadline',
      'value','publicName','boxNo','photoData','urgency'
    ]);
  }

  // 既存行を検索（id一致）
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0] === item.id) {
      // 更新
      sheet.getRange(i + 1, 1, 1, 16).setValues([[
        item.id, item.inquiryDate, item.foundDate, item.place,
        item.item, item.detail, item.contact, item.returnMethod,
        item.urgency, item.status, item.deadline,
        item.value, item.publicName, item.boxNo, item.photoData || '', item.urgency
      ]]);
      return jsonResponse({ success: true, action: 'updated' });
    }
  }

  // 新規追加
  sheet.appendRow([
    item.id, item.inquiryDate, item.foundDate, item.place,
    item.item, item.detail, item.contact, item.returnMethod,
    item.urgency, item.status, item.deadline,
    item.value, item.publicName, item.boxNo, item.photoData || '', item.urgency
  ]);
  return jsonResponse({ success: true, action: 'created' });
}

  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0] === manageId) {
      sheet.getRange(i + 1, 15).setValue(fileUrl); // O列 = photoData
      break;
    }
  }
  return jsonResponse({ success: true, url: fileUrl });
}

// ── doGet：疎通確認 ──
function doGet(e) {
  if (e.parameter.action === 'getAll') {
    return getAllItems();
  }
  return jsonResponse({ status: 'ok', message: '忘れ物管理API稼働中' });
}

// ── レスポンスヘルパー ──
function jsonResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
